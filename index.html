// Function to make the chatbot speak function speakText(text) { const synth = window.speechSynthesis; const utterance = new SpeechSynthesisUtterance(text); utterance.lang = "en-US"; synth.speak(utterance); } // Predefined chatbot messages window.onload = function () { const predefinedMessages = [ "Hello. I am Chitti, the chatbot. Speed 1 THz, Memory 1 zettabyte.", "Welcome to the BCA department chatbot!" ]; const chatLog = document.getElementById("chat-log"); predefinedMessages.forEach((message, index) => { setTimeout(() => { // Add chatbot message chatLog.innerHTML += `
${message}
`; speakText(message); chatLog.scrollTop = chatLog.scrollHeight; }, index * 3000); }); }; // Function to handle sending messages function sendMessage() { const userInput = document.getElementById("user-input").value.trim(); const chatLog = document.getElementById("chat-log"); if (userInput) { // Add user message to the chat log chatLog.innerHTML += `
${userInput}
`; setTimeout(() => { // Keyword-based responses const responses = { hello: "Hi there! How can I assist you?", "how are you": "I'm doing fine! How about you?", name: "I am your friendly BCA department chatbot!", bye: "Goodbye! Have a great day!", bca: "BCA stands for Bachelor of Computer Applications.", help: "Let me know what you need help with!", department: "This chatbot is here to assist with the BCA department queries.", "h.o.d": "Dr. P.K. Rai is the Head of Department." }; // Default response let response = "I'm sorry, I don't understand that."; // Check for keywords in user input for (let keyword in responses) { if (userInput.toLowerCase().includes(keyword)) { response = responses[keyword]; break; } } // Add chatbot response to the chat log chatLog.innerHTML += `
${response}
`; speakText(response); chatLog.scrollTop = chatLog.scrollHeight; }, 1000); // Clear the input field document.getElementById("user-input").value = ''; } } // Function to handle "Enter" key press function handleKeyPress(event) { if (event.key === "Enter") { sendMessage(); } }
